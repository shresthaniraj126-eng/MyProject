-- ============================================================
--  hairdec database schema
-- ============================================================

-- admin
CREATE TABLE admin (
  ADMIN_ID  INT(10)      NOT NULL AUTO_INCREMENT,
  USERNAME  VARCHAR(30)  NOT NULL,
  PASSWORD  VARCHAR(30)  NOT NULL,
  PRIMARY KEY (ADMIN_ID),
  UNIQUE KEY USERNAME (USERNAME)
);

-- users
CREATE TABLE users (
  USERS_ID  INT(10)       NOT NULL AUTO_INCREMENT,
  USERNAME  VARCHAR(30)   NOT NULL,
  PASSWORD  VARCHAR(255)  NOT NULL,
  EMAIL     VARCHAR(50)   NOT NULL,
  TYPE      VARCHAR(20)   NOT NULL,
  PRIMARY KEY (USERS_ID),
  UNIQUE KEY USERNAME (USERNAME),
  UNIQUE KEY EMAIL (EMAIL)
);

-- shop
CREATE TABLE shop (
  SHOP_ID      INT(11)       NOT NULL AUTO_INCREMENT,
  NAME         VARCHAR(100)  NOT NULL,
  DESCRIPTION  TEXT          DEFAULT NULL,
  IMAGES       VARCHAR(255)  DEFAULT NULL,
  USERS_ID     INT(11)       NOT NULL,
  LOCATION     VARCHAR(150)  NOT NULL,
  CONTACT      VARCHAR(20)   NOT NULL,
  SEAT_NO      INT(11)       NOT NULL,
  PRIMARY KEY (SHOP_ID),
  KEY fk_shop_merchant (USERS_ID),
  CONSTRAINT fk_shop_merchant
    FOREIGN KEY (USERS_ID) REFERENCES users (USERS_ID)
);

-- category
CREATE TABLE category (
  CATEGORIES_ID  INT(11)        NOT NULL AUTO_INCREMENT,
  SHOP_ID        INT(11)        NOT NULL,
  NAME           VARCHAR(100)   NOT NULL,
  PRICE          DECIMAL(10,2)  NOT NULL,
  PRIMARY KEY (CATEGORIES_ID),
  KEY fk_category_shop (SHOP_ID),
  CONSTRAINT fk_category_shop
    FOREIGN KEY (SHOP_ID) REFERENCES shop (SHOP_ID)
);

-- booking
CREATE TABLE booking (
  BOOKING_ID  INT(11)        NOT NULL AUTO_INCREMENT,
  USERS_ID    INT(11)        NOT NULL,
  SHOP_ID     INT(11)        NOT NULL,
  TIME        TIME           NOT NULL,
  DATE        DATE           NOT NULL,
  TOTAL       DECIMAL(10,2)  NOT NULL,
  PRIMARY KEY (BOOKING_ID),
  KEY USERS_ID (USERS_ID),
  KEY SHOP_ID (SHOP_ID),
  CONSTRAINT fk_booking_user
    FOREIGN KEY (USERS_ID) REFERENCES users (USERS_ID),
  CONSTRAINT fk_booking_shop
    FOREIGN KEY (SHOP_ID) REFERENCES shop (SHOP_ID)
);

-- wal_log
CREATE TABLE wal_log (
  WAL_ID       INT(11)                                        NOT NULL AUTO_INCREMENT,
  TXN_ID       VARCHAR(64)                                    NOT NULL,
  OPERATION    VARCHAR(50)                                    NOT NULL,
  PAYLOAD      LONGTEXT                                       NOT NULL,
  STATUS       ENUM('PENDING', 'COMMITTED', 'ROLLED_BACK')  DEFAULT 'PENDING',
  CREATED_AT   DATETIME                                       DEFAULT current_timestamp(),
  RESOLVED_AT  DATETIME                                       DEFAULT NULL,
  PRIMARY KEY (WAL_ID)
);
